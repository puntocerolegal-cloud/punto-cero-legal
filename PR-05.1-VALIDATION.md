## 6. Ejemplo Real: OSError DESPUÉS

Con `normalizeHTTPError()` aplicado:

```js
// billing.service.js — createInvoice() ahora:
try {
  const data = normalizeInvoice(unwrap(await apiClient.post("/billing/", payload)));
  eventBus.emit(OS_EVENTS.invoiceCreated, data);
  return data;
} catch (err) {
  normalizeHTTPError(err, {
    service: 'billing',
    operation: 'createInvoice',
    resourceType: 'invoice',
  });
}

// El error que es lanzado ahora es:
OSError {
  name: 'OSError',
  message: 'El estado ha cambiado. Recargue e intente de nuevo.',
  kind: 'conflict',
  statusCode: 409,
  service: 'billing',
  operation: 'createInvoice',
  resourceType: 'invoice',
  resourceId: undefined,
  details: {
    detail: 'La factura con ese número ya existe'
  },
  response: { status: 409, data: { ... }, ... },
  request: XMLHttpRequest { ... },
  config: { ... },
  code: 'ERR_BAD_REQUEST',
  stack: 'Error: Request failed with status code 409\n...'
}
```

**Consumidor ahora puede:**
```js
try {
  await billingService.createInvoice(payload);
} catch (err) {
  // Métodos helper para clasificar sin inspeccionar statusCode
  if (err.isConflict()) {
    console.log(`Conflict on ${err.service}.${err.operation}`);
    // Lógica de reintentos con datos frescos
  } else if (err.isValidation()) {
    console.log(`Validation error: ${err.message}`);
    // Mostrar errores de campo
  } else if (err.isServerError()) {
    console.log(`Server error, contact support`);
    // Mostrar error genérico al usuario
  }

  // Acceso a contexto completo
  console.log({
    timestamp: new Date().toISOString(),
    error: err.message,
    kind: err.kind,
    statusCode: err.statusCode,
    service: err.service,
    operation: err.operation,
    resourceType: err.resourceType,
  });
}
```
