```python
from pybreaker import CircuitBreaker
from tenacity import retry, stop_after_attempt, wait_exponential

class MercadoPagoClient:
    def __init__(self):
        self.breaker = CircuitBreaker(
            fail_max=5,      # Open after 5 failures
            reset_timeout=60, # Try again after 60s
            listeners=[CircuitBreakerListener()]
        )
        self.session = None
    
    @retry(
        stop=stop_after_attempt(3),
        wait=wait_exponential(multiplier=1, min=2, max=10),
        retry=retry_if_exception_type((httpx.TimeoutException, httpx.ConnectionError))
    )
    async def create_preference(self, preference_data):
        if self.breaker.opened:
            raise CircuitBreakerOpen("MercadoPago API is temporarily unavailable")
        
        try:
            async with httpx.AsyncClient(timeout=5) as client:  # Reduced to 5s
                response = await client.post(
                    "https://api.mercadopago.com/checkout/preferences",
                    json=preference_data,
                    headers=self.headers,
                    timeout=5
                )
                response.raise_for_status()
                self.breaker.succeed()
                return response.json()
        except Exception as e:
            self.breaker.fail()
            raise
```

```python
from pymongo.errors import ServerSelectionTimeoutError
import asyncio

async def create_db_connection():
    """Create database connection with fallback"""
    max_retries = 5
    retry_delay = 2
    
    for attempt in range(max_retries):
        try:
            client = AsyncIOMotorClient(
                mongo_url,
                serverSelectionTimeoutMS=5000,  # 5s timeout
                connectTimeoutMS=3000,           # 3s connection timeout
                retryWrites=True,
                retryReads=True,
                maxPoolSize=50,                  # Connection pooling
                minPoolSize=10
            )
            
            # Test connection
            await client.admin.command('ping')
            
            db = create_guarded_db(client[DB_NAME])
            logger.info("MongoDB connected successfully")
            return db
            
        except ServerSelectionTimeoutError as e:
            if attempt < max_retries - 1:
                logger.warning(f"DB connection attempt {attempt+1} failed, retrying...")
                await asyncio.sleep(retry_delay * (2 ** attempt))  # Exponential backoff
            else:
                logger.error("Failed to connect to MongoDB after retries, using fallback")
                return create_fallback_db()  # Use in-memory DB
        except Exception as e:
            logger.error(f"Unexpected error: {e}")
            return create_fallback_db()

# At startup
app.state.db = await create_db_connection()

    except Exception as e:
```

```python
MAX_QUERY_TIMEOUT_MS = 5000  # 5 second timeout
MAX_RESULT_SIZE = 10000      # Maximum 10k results

async def safe_find_all(collection, query, timeout_ms=MAX_QUERY_TIMEOUT_MS, limit=MAX_RESULT_SIZE):
    """Find with timeout and limit"""
    try:
        cursor = collection.find(query).max_time_ms(timeout_ms).limit(limit)
```

### FINDING #4: NO GRACEFUL SHUTDOWN HANDLING
**File:** `backend/server.py` (shutdown logic missing)
**Severity:** CRITICAL
**Impact:** Data loss, corrupted transactions

```python
# Current: FastAPI app just stops on SIGTERM
# No cleanup of:
# - Active database connections
# - Pending webhook processing
# - Cron jobs
# - In-flight transactions
```

**Production Impact:**
```
1. Kubernetes sends SIGTERM to pod (30s grace period)
2. Active payment webhook being processed (no transaction)
3. Process killed mid-operation
4. Payment status lost, money not recorded
5. Customer never gets credited
```

**Required Implementation:**

```python
import asyncio
from contextlib import asynccontextmanager

class GracefulShutdown:
    def __init__(self):
        self.active_tasks = set()
        self.shutdown_event = asyncio.Event()
    
    async def track_task(self, coro):
        """Track long-running operations"""
        task = asyncio.create_task(coro)
        self.active_tasks.add(task)
        task.add_done_callback(self.active_tasks.discard)
        return await task
    
    async def shutdown(self):
        """Graceful shutdown"""
        logger.info("Initiating graceful shutdown...")
        self.shutdown_event.set()
        
        # Give tasks 15 seconds to complete
        try:
            await asyncio.wait_for(
                asyncio.gather(*self.active_tasks, return_exceptions=True),
                timeout=15
            )
        except asyncio.TimeoutError:
            logger.warning("Forced shutdown after timeout")
        
        # Close database connections
        if app.state.db:
            app.state.db.client.close()
        
        logger.info("Shutdown complete")

shutdown_handler = GracefulShutdown()

@asynccontextmanager
async def lifespan(app: FastAPI):
    # Startup
    app.state.cron_jobs = CronJobManager(app.state.db)
    await app.state.cron_jobs.start()
    yield
    # Shutdown
    await shutdown_handler.shutdown()

app = FastAPI(lifespan=lifespan)

# Handle signals
import signal
def handle_sigterm(signum, frame):
    asyncio.create_task(shutdown_handler.shutdown())

signal.signal(signal.SIGTERM, handle_sigterm)
signal.signal(signal.SIGINT, handle_sigterm)
```

**Required Implementation:**

```python
async def process_payment_webhook_idempotent(db, webhook_data):
    """Process webhook with idempotency guarantee"""
    event_id = webhook_data["id"]  # Unique webhook ID
    
    async with await db.client.start_session() as session:
        async with session.start_transaction():
            # Step 1: Check if we've processed this webhook
            processed = await db.webhook_events.find_one(
                {"event_id": event_id, "status": "processed"},
                session=session
            )
            if processed:
                logger.info(f"Webhook {event_id} already processed, skipping")
                return  # Already handled
            
            # Step 2: Record that we're processing
            await db.webhook_events.update_one(
                {"event_id": event_id},
                {"$set": {"status": "processing", "started_at": datetime.utcnow()}},
                upsert=True,
                session=session
            )
            
            try:
                # Step 3: All payment operations in same transaction
                payment_id = webhook_data["data"]["id"]
                
                tx = await db.transactions.find_one(
                    {"payment_id": payment_id},
                    session=session
                )
                
                if not tx:
                    raise ValueError(f"Transaction not found: {payment_id}")
                
                # Mark as paid
                await db.transactions.update_one(
                    {"_id": tx["_id"]},
                    {"$set": {"status": "paid", "paid_at": datetime.utcnow()}},
                    session=session
                )
                
                # Update user
                await db.users.update_one(
                    {"email": tx["user_email"]},
                    {"$set": {"subscription_status": "active"}},
                    session=session
                )
                
                # Record success
                await db.webhook_events.update_one(
                    {"event_id": event_id},
                    {"$set": {"status": "processed", "completed_at": datetime.utcnow()}},
                    session=session
                )
                
            except Exception as e:
                # Record failure
                await db.webhook_events.update_one(
                    {"event_id": event_id},
                    {"$set": {"status": "failed", "error": str(e)}},
                    session=session
                )
                raise
```

```python
import asyncio

class CronJobManager:
    def __init__(self, db):
        self.db = db
        self.running_jobs = set()
    
    async def run_daily_task(self, job_name):
        # Use distributed lock
        lock_acquired = await self._acquire_distributed_lock(job_name)
        if not lock_acquired:
            logger.info(f"Job {job_name} already running, skipping")
            return
        
        try:
            await self._execute_job(job_name)
        finally:
            await self._release_distributed_lock(job_name)
    
    async def _acquire_distributed_lock(self, job_name):
        """Use MongoDB as distributed lock"""
        result = await self.db.cron_locks.find_one_and_update(
            {"job": job_name},
            {"$set": {"locked_at": datetime.utcnow(), "expires_at": datetime.utcnow() + timedelta(hours=1)}},
            upsert=True,
            return_document=ReturnDocument.BEFORE
        )
        # Lock acquired if previous value was None or expired
        return result is None or result.get("expires_at", datetime.min) < datetime.utcnow()
```
