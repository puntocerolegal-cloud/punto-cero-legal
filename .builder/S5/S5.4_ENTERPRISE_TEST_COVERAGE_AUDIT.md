import pytest
from unittest.mock import AsyncMock, patch
import hmac
import hashlib

@pytest.mark.asyncio
async def test_webhook_hmac_validation():
    """Verify webhook signature is validated"""
    payload = {"id": 123, "status": "approved"}
    secret = "test_secret"
    
    # Calculate valid HMAC
    signature = hmac.new(
        secret.encode(),
        json.dumps(payload).encode(),
        hashlib.sha256
    ).hexdigest()
    
    # Send webhook with valid signature
    response = await client.post(
        "/api/payment/webhook",
        json=payload,
        headers={"X-Signature": signature}
    )
    assert response.status_code == 200

@pytest.mark.asyncio
async def test_webhook_duplicate_rejection():
    """Verify duplicate webhooks are rejected"""
    payload = {"id": 123, "status": "approved", "time": "2024-01-01T00:00:00Z"}
    
    # First webhook processes
    response1 = await client.post("/api/payment/webhook", json=payload)
    assert response1.status_code == 200
    
    # Second identical webhook is rejected
    response2 = await client.post("/api/payment/webhook", json=payload)
    assert response2.status_code == 409  # Conflict - already processed

@pytest.mark.asyncio
